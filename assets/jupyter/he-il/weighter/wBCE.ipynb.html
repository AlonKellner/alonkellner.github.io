def BCE(y, y_hat):
    return y_hat, -y * np.log(y_hat) - (1 - y) * np.log(1 - y_hat)


def wBCE(y, y_hat):
    return y_hat, -y * np.log(y_hat) + y_hat

def d_dyhat(f):
    def d_dyhat_f(y, y_hat):
        f_x, f_y = f(y, y_hat)
        return f_x, np.diff(f_y)/np.diff(f_x)
    return d_dyhat_f
